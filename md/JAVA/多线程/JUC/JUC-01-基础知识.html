# JUC基础知识回顾

wait / sleep 的区别

1、来自不同的类 这两个方法来自不同的类分别是，sleep来自Thread类，和wait来自Object类。

2、有没有释放锁(释放资源) 最主要是sleep方法没有释放锁，而wait方法释放了锁，使得其他线程可以使用同步控制块或者方法。

3、使用范围不同 wait，notify和notifyAll只能在同步控制方法或者同步控制块里面使用，而sleep可以在任何地方使用

synchronized 和 lock 区别

1、Lock是java的一个interface接口，而synchronized是Java中的关键字，synchronized是由JDK实现的，不需要程序员编写代码去控制加锁和释放；Lock的接口如下：

public interface Lock {
    void lock();
    void lockInterruptibly() throws InterruptedException;
    boolean tryLock();
    boolean tryLock(long time, TimeUnit unit) throws InterruptedException;
    void unlock();
    Condition newCondition();
}

2、synchronized修饰的代码在执行异常时，jdk会自动释放线程占有的锁，不需要程序员去控制释放锁，因此不会导致死锁现象发生；但是，当Lock发生异常时，如果程序没有通过unLock()去释放锁，则很可能造成死锁现象，因此Lock一般都是在finally块中释放锁；格式如下：

Lock lock = new LockImpl; // new 一个Lock的实现类
lock.lock();  // 加锁
try{
    //todo
}catch(Exception ex){
     // todo
}finally{
    lock.unlock();   //释放锁
}

3、Lock可以让等待锁的线程响应中断处理，如tryLock(long time, TimeUnit unit)，而synchronized却不行，使用synchronized时，等待的线程会一直等待下去，不能够中断，程序员无法控制； 4、synchronized是非公平锁，Lock可以设置是否公平锁，默认是非公平锁； 5、Lock的实现类ReentrantReadWriteLock提供了readLock()和writeLock()用来获取读锁和写锁的两个方法，这样多个线程可以进行同时读操作； 6、Lock锁的范围有局限性，仅适用于代码块范围，而synchronized可以锁住代码块、对象实例、类； 7、Lock可以绑定条件，实现分组唤醒需要的线程；synchronized要么随机唤醒一个，要么唤醒全部线程。

# synchronized实现

要用while做条件判断，防止虚假唤醒

package com.example.jucdemo.testdemo.pac;

/**
 * 题目：现在四个线程，可以操作初始值为0的一个变量
 * 实现两个线程对该变量 + 1，两个线程对该变量 -1
 * 实现交替10次
 * <p>
 * 诀窍：
 * 1. 高内聚低耦合的前提下，线程操作资源类
 * 2. 判断 、干活、通知
 * 3. 多线程交互中，必须要防止多线程的虚假唤醒，也即（判断不能用if，只能用while）
 */
public class synchronizedImpl {
    public static void main(String[] args) {
        Date date = new Date();
        new Thread(() -> {
            for (int i = 1; i <= 10; i++) {
                try {
                    date.increment();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        },"A").start();
        new Thread(() -> {
            for (int i = 1; i <= 10; i++) {
                try {
                    date.increment();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        },"B").start();
        new Thread(() -> {
            for (int i = 1; i <= 10; i++) {
                try {
                    date.decrement();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        },"C").start();
        new Thread(() -> {
            for (int i = 1; i <= 10; i++) {
                try {
                    date.decrement();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        },"D").start();
    }


}

class Date {
    private int num = 0;

    //+1
    public synchronized void increment() throws InterruptedException {
        while (num != 0) {
            wait();
        }
        num++;
        System.out.println(Thread.currentThread().getName()+":"+num);
        notifyAll();
    }

    //-1
    public synchronized void decrement() throws InterruptedException {
        while (num == 0) {
            wait();
        }
        num--;
        System.out.println(Thread.currentThread().getName()+":"+num);
        notifyAll();
    }
}

# Lock实现

package com.example.jucdemo.testdemo.pac;

import java.util.concurrent.locks.Condition;
import java.util.concurrent.locks.Lock;
import java.util.concurrent.locks.ReentrantLock;

/**
 * 题目：现在四个线程，可以操作初始值为0的一个变量
 * 实现两个线程对该变量 + 1，两个线程对该变量 -1
 * 实现交替10次
 * <p>
 * 诀窍：
 * 1. 高内聚低耦合的前提下，线程操作资源类
 * 2. 判断 、干活、通知
 * 3. 多线程交互中，必须要防止多线程的虚假唤醒，也即（判断不能用if，只能用while）
 */
public class LockImpl {
    public static void main(String[] args) {
        Date1 date = new Date1();
        new Thread(() -> { for (int i = 1; i <= 10; i++)date.increment(); },"A").start();
        new Thread(() -> { for (int i = 1; i <= 10; i++)date.increment(); },"B").start();
        new Thread(() -> { for (int i = 1; i <= 10; i++)date.decrement(); },"C").start();
        new Thread(() -> { for (int i = 1; i <= 10; i++)date.decrement(); },"D").start();
    }
}
class Date1{
    private int num=0;
    private Lock lock=new ReentrantLock();
    private Condition condition=lock.newCondition();
    //+1
    public void increment(){
        lock.lock();
        try {
            while(num!=0){
                condition.await();
            }
            num++;
            System.out.println(Thread.currentThread().getName()+":"+num);
            condition.signalAll();
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            lock.unlock();
        }
    }

    //-1
    public void decrement() {
        lock.lock();
        try {
            while(num==0){
                condition.await();
            }
            num--;
            System.out.println(Thread.currentThread().getName()+":"+num);
            condition.signalAll();
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            lock.unlock();
        }
    }
}

# 精确通知

使用Condition的await和singal方法去精准通知。

package com.example.jucdemo.testdemo.pac;

import java.util.concurrent.locks.Condition;
import java.util.concurrent.locks.Lock;
import java.util.concurrent.locks.ReentrantLock;

public class LockAdviceImpl {
    public static void main(String[] args) {
        Data2 data = new Data2();
        new Thread(()->{for (int i = 1; i <=10 ; i++) data.increment1(); },"A").start();
        new Thread(()->{for (int i = 1; i <=10 ; i++) data.increment2(); },"B").start();
        new Thread(()->{for (int i = 1; i <=10 ; i++) data.increment3(); },"C").start();
    }
}

class Data2{
    private int num=0;
    private Lock lock=new ReentrantLock();
    private Condition condition1=lock.newCondition();
    private Condition condition2=lock.newCondition();
    private Condition condition3=lock.newCondition();
    //+1
    public void increment1(){
        lock.lock();
        try {
            while (num!=0){
                condition1.await();
            }
            num++;
            System.out.println(Thread.currentThread().getName()+":"+num);
            condition2.signal();
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            lock.unlock();
        }
    }
    //+1
    public void increment2(){
        lock.lock();
        try {
            while (num!=1){
                condition2.await();
            }
            num++;
            System.out.println(Thread.currentThread().getName()+":"+num);
            condition3.signal();
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            lock.unlock();
        }
    }
    //+1
    public void increment3(){
        lock.lock();
        try {
            while (num!=2){
                condition3.await();
            }
            num=0;
            System.out.println(Thread.currentThread().getName()+":"+num);
            condition1.signal();
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            lock.unlock();
        }
    }
}
