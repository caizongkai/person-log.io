# 隐藏Tomcat版本号

如果tomcat的版本号暴露出来，会有人利用该版本所存在的漏洞进行攻击 隐藏tomcat 版本号 1.进入tomcat的lib目录找到catalina.jar文件

cd /home/bmsoft/tomcat/tomcat-
1
2
unzip catalina.jar
# 之后会多出两个文件夹，进入org/apache/catalina/util 
cd org/apache/catalina/util 
# 编辑配置文件ServerInfo.properties
vim ServerInfo.properties
1
2
3
4
5

修改为

server.info=Apache Tomcat
server.number=0.0.0.0
server.built=Nov 7 2016 20:05:27 UTC
1
2
3

将修改后的信息压缩回jar包

cd  /tomcat/lib
cd ../../../..
jar uvf catalina.jar org/apache/catalina/util/ServerInfo.properties

# 隐藏nginx版本号

[root@proxy ~]# vim /usr/local/nginx/conf/nginx.conf
http {
    include       mime.types;
    default_type  application/octet-stream;
 
    server_tokens off;
    #### 在http节点下加入server_tokens off;即可
